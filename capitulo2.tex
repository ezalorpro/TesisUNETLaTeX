							k1 = h * (ss.A * x + ss.B * u)
							k2 = h * (ss.A * (x+k1/2) + ss.B * u)
							k3 = h * (ss.A * (x+k2/2) + ss.B * u)

							x = x + (1/6)*(k1 + 2*k2 + 2*k3 + k4)
